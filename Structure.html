Array Examples

1. Find the largest element in an array.


#include <stdio.h>

int findLargest(int arr[], int size);

int main() {
    int arr[] = {1, 5, 3, 9, 2};
    int size = sizeof(arr) / sizeof(arr[0]);
    int largest = findLargest(arr, size);
    printf("Largest element is %d\n", largest);
    return 0;
}

int findLargest(int arr[], int size) {
    int max = arr[0];
    for (int i = 1; i < size; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    return max;
}

2. Find the sum of elements in an array.


#include <stdio.h>

int findSum(int arr[], int size);

int main() {
    int arr[] = {1, 2, 3, 4, 5};
    int size = sizeof(arr) / sizeof(arr[0]);
    int sum = findSum(arr, size);
    printf("Sum of elements is %d\n", sum);
    return 0;
}

int findSum(int arr[], int size) {
    int sum = 0;
    for (int i = 0; i < size; i++) {
        sum += arr[i];
    }
    return sum;
}

3. Reverse the array.


#include <stdio.h>

void reverseArray(int arr[], int size);

int main() {
    int arr[] = {1, 2, 3, 4, 5};
    int size = sizeof(arr) / sizeof(arr[0]);
    reverseArray(arr, size);
    printf("Reversed array: ");
    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
    return 0;
}

void reverseArray(int arr[], int size) {
    int start = 0, end = size - 1;
    while (start < end) {
        int temp = arr[start];
        arr[start] = arr[end];
        arr[end] = temp;
        start++;
        end--;
    }
}

4. Search for an element in an array.


#include <stdio.h>

int searchElement(int arr[], int size, int key);

int main() {
    int arr[] = {1, 2, 3, 4, 5};
    int size = sizeof(arr) / sizeof(arr[0]);
    int key = 3;
    int index = searchElement(arr, size, key);
    if (index != -1)
        printf("Element %d found at index %d\n", key, index);
    else
        printf("Element %d not found\n", key);
    return 0;
}

int searchElement(int arr[], int size, int key) {
    for (int i = 0; i < size; i++) {
        if (arr[i] == key) {
            return i;
        }
    }
    return -1;
}

5. Sort the array in ascending order.


#include <stdio.h>

void sortArray(int arr[], int size);

int main() {
    int arr[] = {5, 2, 9, 1, 5, 6};
    int size = sizeof(arr) / sizeof(arr[0]);
    sortArray(arr, size);
    printf("Sorted array: ");
    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
    return 0;
}

void sortArray(int arr[], int size) {
    for (int i = 0; i < size - 1; i++) {
        for (int j = i + 1; j < size; j++) {
            if (arr[i] > arr[j]) {
                int temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
}

Structure Examples

1. Define and use a simple structure.


#include <stdio.h>

struct Point {
    int x;
    int y;
};

int main() {
    struct Point p1;
    p1.x = 10;
    p1.y = 20;
    printf("Point coordinates: (%d, %d)\n", p1.x, p1.y);
    return 0;
}

2. Structure with nested structures.


#include <stdio.h>

struct Date {
    int day;
    int month;
    int year;
};

struct Employee {
    char name[50];
    struct Date dob;
};

int main() {
    struct Employee emp = {"John Doe", {15, 4, 1990}};
    printf("Employee: %s\nDate of Birth: %02d/%02d/%d\n", emp.name, emp.dob.day, emp.dob.month, emp.dob.year);
    return 0;
}

3. Structure with pointers.


#include <stdio.h>

struct Person {
    char name[50];
    int age;
};

void displayPerson(struct Person *p);

int main() {
    struct Person p1 = {"Alice", 30};
    displayPerson(&p1);
    return 0;
}

void displayPerson(struct Person *p) {
    printf("Name: %s\nAge: %d\n", p->name, p->age);
}

#include <stdio.h>

5. Structure with functions.


#include <stdio.h>

struct Rectangle {
    int width;
    int height;
};

int calculateArea(struct Rectangle *r);

int main() {
    struct Rectangle rect = {10, 20};
    int area = calculateArea(&rect);
    printf("Area of the rectangle: %d\n", area);
    return 0;
}

int calculateArea(struct Rectangle *r) {
    return r->width * r->height;
}

Union Examples

1. Define and use a simple union.


#include <stdio.h>

union Data {
    int intValue;
    float floatValue;
    char charValue;
};

int main() {
    union Data data;
    data.intValue = 10;
    printf("Integer: %d\n", data.intValue);
    data.floatValue = 220.5;
    printf("Float: %.2f\n", data.floatValue);
    data.charValue = 'A';
    printf("Character: %c\n", data.charValue);
    return 0;
}

2. Union with different data types.


#include <stdio.h>

union Value {

int main() {

    v.f = 15.5;
    printf("Float value: %.2f\n", v.f);
    v.c = 'X';

4. Union and structure together.


#include <stdio.h>

struct Info {
    union {
        int id;
        float salary;
    } data;
    char name[50];
};

int main() {
    struct Info emp;
    emp.data.salary = 50000.75;
    snprintf(emp.name, sizeof(emp.name), "John Doe");
    printf("Name: %s\nSalary: %.2f\n", emp.name, emp.data.salary);
    return 0;
}

5. Size of union and its elements.


#include <stdio.h>

union Data {
    int i;
    float f;
    char c;
};

int main() {
    union Data data;
    printf("Size of union: %zu bytes\n", sizeof(data));
    printf("Size of int in union: %zu bytes\n", sizeof(data.i));
    printf("Size of float in union: %zu bytes\n", sizeof(data.f));
    printf("Size of char in union: %zu bytes\n", sizeof(data.c));
    return 0;
}
