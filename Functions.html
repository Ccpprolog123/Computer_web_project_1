1. Write a C program to find the square of a given number using user-defined functions.


#include <stdio.h>

void takeInput(int *num);
int calculateSquare(int num);
void displayResult(int result);

int main() {
    int num, square;
    takeInput(&num);
    square = calculateSquare(num);
    displayResult(square);
    return 0;
}

void takeInput(int *num) {
    printf("Enter a number: ");
    scanf("%d", num);
}

int calculateSquare(int num) {
    return num * num;
}

void displayResult(int result) {
    printf("Square: %d\n", result);
}

2. Write a C program to calculate the area of the circle using a function.

7. Write a C program to swap two numbers using a function.


#include <stdio.h>

void swap(int *a, int *b);

int main() {
    int a, b;
    printf("Enter two numbers: ");
    scanf("%d %d", &a, &b);
    swap(&a, &b);
    printf("After swapping: a = %d, b = %d\n", a, b);
    return 0;
}

void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

9. Write a C program to check if the input number is perfect or not.


#include <stdio.h>

int isPerfect(int num);

int main() {
    int num;
    printf("Enter a number: ");
    scanf("%d", &num);
    if (isPerfect(num))
        printf("%d is a perfect number\n", num);
    else
        printf("%d is not a perfect number\n", num);
    return 0;
}

int isPerfect(int num) {
    int sum = 0;
    for (int i = 1; i <= num / 2; i++) {
        if (num % i == 0) {
            sum += i;
        }
    }
    return (sum == num);
}

10. Write a C program to find the maximum and minimum values among three numbers. Use two separate functions for maximum and minimum.


#include <stdio.h>

int findMax(int a, int b, int c);
int findMin(int a, int b, int c);

int main() {
    int a, b, c;
    printf("Enter three numbers: ");
    scanf("%d %d %d", &a, &b, &c);
    printf("Maximum value: %d\n", findMax(a, b, c));
    printf("Minimum value: %d\n", findMin(a, b, c));
    return 0;
}

int findMax(int a, int b, int c) {
    if (a > b && a > c) return a;
    if (b > a && b > c) return b;
    return c;
}

int findMin(int a, int b, int c) {
    if (a < b && a < c) return a;
    if (b < a && b < c) return b;
    return c;
}

II. Recursive Function Examples

1. Write a C program using recursive function to find the factorial of a given number.


#include <stdio.h>

int factorial(int n);

int main() {
    int n;
    printf("Enter a positive integer: ");
    scanf("%d", &n);
    printf("Factorial: %d\n", factorial(n));
    return 0;
}

int factorial(int n) {
    if (n == 0)
        return 1;
    return n * factorial(n - 1);
}

void fibonacci(int n, int a, int b) {

        fibonacci(n - 1, b, a + b);
    }
}

3. Write a C program using recursive function to check if the given number is palindrome or not.


#include <stdio.h>

int isPalindrome(int num, int reversed);

int main() {
    int num;
    printf("Enter a number: ");
    scanf("%d", &num);
    int reversed = 0;
    if (isPalindrome(num, reversed))
        printf("%d is a Palindrome number\n", num);
    else
        printf("%d is not a Palindrome number\n", num);
    return 0;
}

int isPalindrome(int num, int reversed) {
    if (num == 0)
        return (reversed == 0);
    reversed = reversed * 10 + num % 10;
    return isPalindrome(num / 10, reversed);
}

6. Find the Greatest Common Divisor (GCD) of two numbers using a function.


#include <stdio.h>

int gcd(int a, int b);

int main() {
    int a, b;
    printf("Enter two numbers: ");
    scanf("%d %d", &a, &b);
    printf("GCD: %d\n", gcd(a, b));
    return 0;
}

int gcd(int a, int b) {
    if (b == 0)
        return a;
    return gcd(b, a % b);
}

7. Check whether the given number is Prime or not using a function.


#include <stdio.h>

int isPrime(int num);

int main() {
    int num;
    printf("Enter a number: ");
    scanf("%d", &num);
    if (isPrime(num))
        printf("%d is a prime number\n", num);
    else
        printf("%d is not a prime number\n", num);
    return 0;
}

int isPrime(int num) {
    if (num <= 1)
        return 0;
    for (int i = 2; i <= num / 2; i++) {
        if (num % i == 0)
            return 0;
    }
    return 1;
}

8. Count the number of digits of the input number.

    int num;

    scanf("%d", &num);
    printf("Number of digits: %d\n", countDigits(num));
    return 0;
}

int countDigits(int num) {
    int count = 0;
    while (num != 0) {
        num /= 10;
        count++;
    }
    return count;
}

12. Sort the elements of the user input array in ascending order using a function.


#include <stdio.h>

void sortArray(int arr[], int size);

int main() {
    int arr[5];
    printf("Enter 5 integers:\n");
    for (int i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
    }
    sortArray(arr, 5);
    printf("Sorted array: ");
    for (int i = 0; i < 5; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
    return 0;
}

void sortArray(int arr[], int size) {
    for (int i = 0; i < size - 1; i++) {
        for (int j = i + 1; j < size; j++) {
            if (arr[i] > arr[j]) {
                int temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
}
