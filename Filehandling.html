1. Write a C program to write a character and display it using fgetc() and fputc() functions.


#include <stdio.h>

int main() {
    FILE *file;
    char ch;

    file = fopen("test.txt", "w"); // Open file in write mode
    if (file == NULL) {
        printf("Unable to open file.\n");
        return 1;
    }

    fputc('A', file); // Write a character to the file
    fclose(file);

    file = fopen("test.txt", "r"); // Open file in read mode
    if (file == NULL) {
        printf("Unable to open file.\n");
        return 1;
    }

    ch = fgetc(file); // Read a character from the file
    printf("Character read from file: %c\n", ch);
    fclose(file);

    return 0;
}

2. Write a C program to write a number and display it using getw() and putw() functions.


#include <stdio.h>

int main() {
    FILE *file;
    int num;

    file = fopen("number.txt", "w"); // Open file in write mode
    if (file == NULL) {
        printf("Unable to open file.\n");
        return 1;
    }

    putw(12345, file); // Write an integer to the file
    fclose(file);

    file = fopen("number.txt", "r"); // Open file in read mode
    if (file == NULL) {
        printf("Unable to open file.\n");
        return 1;
    }

    num = getw(file); // Read an integer from the file
    printf("Number read from file: %d\n", num);
    fclose(file);

    return 0;
}

3. Write a C program to input a string and display it using fputs() and fgets() functions.


#include <stdio.h>

int main() {
    FILE *file;
    char str[100];

    file = fopen("string.txt", "w"); // Open file in write mode
    if (file == NULL) {
        printf("Unable to open file.\n");
        return 1;
    }

    fputs("Hello, World!", file); // Write a string to the file
    fclose(file);

    file = fopen("string.txt", "r"); // Open file in read mode
    if (file == NULL) {
        printf("Unable to open file.\n");
        return 1;
    }

    fgets(str, sizeof(str), file); // Read a string from the file
    printf("String read from file: %s\n", str);
    fclose(file);

    return 0;
}

4. Write a program to copy the contents of one file to another.


#include <stdio.h>

int main() {
    FILE *source, *destination;
    char ch;

    source = fopen("source.txt", "r"); // Open source file in read mode
    if (source == NULL) {
        printf("Unable to open source file.\n");
        return 1;
    }

    destination = fopen("destination.txt", "w"); // Open destination file in write mode
    if (destination == NULL) {
        printf("Unable to open destination file.\n");
        fclose(source);
        return 1;
    }

    while ((ch = fgetc(source)) != EOF) {
        fputc(ch, destination); // Copy character by character
    }

    fclose(source);
    fclose(destination);

    printf("File copied successfully.\n");
    return 0;
}

5. Write a program to count the number of characters in a file you have created.


#include <stdio.h>

int main() {
    FILE *file;
    int count = 0;
    char ch;

    file = fopen("file.txt", "r"); // Open file in read mode
    if (file == NULL) {
        printf("Unable to open file.\n");
        return 1;
    }

    while ((ch = fgetc(file)) != EOF) {
        count++; // Count characters
    }

    fclose(file);

    printf("Number of characters in the file: %d\n", count);
    return 0;
}

6. Write a C program to read character of 10th position and display it using fseek() and ftell() functions.


#include <stdio.h>

int main() {
    FILE *file;
    char ch;
    long pos;

    file = fopen("file.txt", "r"); // Open file in read mode
    if (file == NULL) {
        printf("Unable to open file.\n");
        return 1;
    }

    fseek(file, 9, SEEK_SET); // Move to the 10th character (index 9)
    pos = ftell(file); // Get current position

    ch = fgetc(file); // Read the character at the current position
    printf("Character at position %ld: %c\n", pos + 1, ch);
    
    fclose(file);
    return 0;
}
