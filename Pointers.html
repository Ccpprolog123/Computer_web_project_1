1. Write a C program to calculate the area of a circle using Pointer.


#include <stdio.h>

void calculateArea(float radius, float *area);

int main() {
    float radius, area;
    printf("Enter the radius of the circle: ");
    scanf("%f", &radius);
    calculateArea(radius, &area);
    printf("Area of the circle: %.2f\n", area);
    return 0;
}

void calculateArea(float radius, float *area) {
    *area = 3.14 * radius * radius;
}

2. Write a C program to find the maximum value between two numbers using Pointer.


#include <stdio.h>

void findMax(int *a, int *b, int *max);

int main() {
    int a, b, max;
    printf("Enter two numbers: ");
    scanf("%d %d", &a, &b);
    findMax(&a, &b, &max);
    printf("Maximum value: %d\n", max);
    return 0;
}

void findMax(int *a, int *b, int *max) {
    *max = (*a > *b) ? *a : *b;
}

3. Write a C program to swap two numbers using call by reference.


#include <stdio.h>

void swap(int *a, int *b);

int main() {
    int a, b;
    printf("Enter two numbers: ");
    scanf("%d %d", &a, &b);
    swap(&a, &b);
    printf("After swapping: a = %d, b = %d\n", a, b);
    return 0;
}

void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

5. Write a C program to check whether the input number is prime or not using Pointer and Function.


#include <stdio.h>

int isPrime(int *num);

int main() {
    int num;
    printf("Enter a number: ");
    scanf("%d", &num);
    if (isPrime(&num))
        printf("%d is a prime number\n", num);
    else
        printf("%d is not a prime number\n", num);
    return 0;
}

int isPrime(int *num) {
    if (*num <= 1)
        return 0;
    for (int i = 2; i <= *num / 2; i++) {
        if (*num % i == 0)
            return 0;
    }
    return 1;
}

6. Write a C program to find the factorial of a given number using call by reference.


#include <stdio.h>

void factorial(int *n, int *result);

int main() {
    int n, result;
    printf("Enter a positive integer: ");
    scanf("%d", &n);
    factorial(&n, &result);
    printf("Factorial: %d\n", result);
    return 0;
}

void factorial(int *n, int *result) {
    *result = 1;
    for (int i = 1; i <= *n; i++) {
        *result *= i;
    }
}

7. Write a C program to print the array of 5 numbers in ascending order using Pointer.


#include <stdio.h>

void sortArray(int *arr, int size);

int main() {
    int arr[5];
    printf("Enter 5 numbers:\n");
    for (int i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
    }
    sortArray(arr, 5);
    printf("Array in ascending order:\n");
    for (int i = 0; i < 5; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
    return 0;
}

void sortArray(int *arr, int size) {
    for (int i = 0; i < size - 1; i++) {
        for (int j = i + 1; j < size; j++) {
            if (*(arr + i) > *(arr + j)) {
                int temp = *(arr + i);
                *(arr + i) = *(arr + j);
                *(arr + j) = temp;
            }
        }
    }
}

8. Write a C program to calculate the length of string using Pointer and Function.


#include <stdio.h>

int stringLength(char *str);

int main() {
    char str[100];
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    printf("Length of the string: %d\n", stringLength(str));
    return 0;
}

int stringLength(char *str) {
    int length = 0;
    while (*str != '\0') {
        length++;
        str++;
    }
    return length - 1; // Subtract 1 to exclude the newline character if present
}

9. Write a C program to find the reverse of a string using call by reference.


#include <stdio.h>
#include <string.h>

void reverseString(char *str);

int main() {
    char str[100];
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    reverseString(str);
    printf("Reversed string: %s\n", str);
    return 0;
}

void reverseString(char *str) {
    int length = strlen(str) - 1; // Exclude the newline character
    for (int i = 0; i < length / 2; i++) {
        char temp = str[i];
        str[i] = str[length - i - 1];
        str[length - i - 1] = temp;
    }
}

10. Write a C program to search an element in an array using pointers.


#include <stdio.h>

int searchElement(int *arr, int size, int key);

int main() {
    int arr[5], key, found;
    printf("Enter 5 numbers:\n");
    for (int i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
    }
    printf("Enter the number to search: ");
    scanf("%d", &key);
    found = searchElement(arr, 5, key);
    if (found != -1)
        printf("Element found at index %d\n", found);
    else
        printf("Element not found\n");
    return 0;
}

int searchElement(int *arr, int size, int key) {
    for (int i = 0; i < size; i++) {
        if (*(arr + i) == key) {
            return i;
        }
    }
    return -1;
}
